控制器动作（Action）

        
         Ocara打破了传统框架的规矩，默认将Action独立成类，但是仍然可以将Action方法写在控制器。两者并存时优先使用后者。
		 

第1种 Action独立成类（默认、推荐）


  默认方式。适合于不另外写业务逻辑类，或者一个Action可以同时处理显示页面、表单提交或Ajax等操作，而不是分成多个Action处理。

特色：

每个Action类都必须继承它的控制器类。类名和文件名统一格式为：首字母大写的动作名称+Action。如IndexAction		 
    实例如下。

use app\controller\item\ItemController;
use app\dal\models\defaults\User;

class EditAction extends ItemController
{
	/**
	 * 初始化代码
	 */
	public function __action()
	{
	    //初始化代码
	}

    public function display()
	{
	 	$userId = $this->request->getGet('user_id');

	    //获取用户信息
	    $user = User::select($userId);

	    if (empty($user)) {
	        ocError('用户不存在或已删除！');
	    }

	    $data = ->toArray();
	    $this->assign('data', $data);
	}

	public function _from()
	{
	 	$userId = $this->request->getGet('user_id');
	    $url = ocUrl('item/update', array('user_id' => $userId);

	    //表单生成代码
        $this->form('form_update')
	         ->init($url);
	}

	public function submit()
	{
	    //更新用户信息
	    $userId = $this->request->getGet('user_id');
	    $posts = $this->request->getPost();
	    $user = User::select($userId);

	    if (empty($user)) {
	        ocError('用户不存在或已删除！');
	    }

        $user->update_time = time();
	    $user->save($posts); //会过滤掉主键user_id字段和不存在的字段
	}
}
 

     （2）优化代码
             以上是直接在控制器中写代码，代码可能太多，所以可以另写业务逻辑类。放在模块目录下面的service下面：

/**
 * User业务逻辑类
 */
namespace service;

use Base\BaseService;
use app\dal\models\defaults\User; //引入User模型类

class UserService extends BaseService
{
	/**
	 * 获取用户信息
	 */
	public function getUserInfo($userId)
	{
		$user = User::select($userId);

	    if (empty($user)) {
	        ocError('用户不存在或已删除！');
	    }

	    $data = $user->toArray();
		return $data;
	}

	/**
	 * 更新用户信息
	 */
	public function editUser($request)
	{
		$user = User::select($request['user_id']);

		if (empty($user)) {
	        ocError('用户不存在或已删除！');
	    }

		$user->update_time = time();
		$user->save($request); //会过滤掉主键user_id字段和不存在的字段
	}
}

Action类的结构控制函数介绍


     Action是独立成类的代码更加美观和易于维护，因为在该类中提供了一些结构函数：
     （1）__action()
             用于写初始化代码（类似于类的析构函数）。这个方法会首先执行，因此不需要新建_construct()函数，以免覆盖父级类的析构函数。
             如下实例：

	protected function __action()
	{
		$this->model = $this->model();
	}


     （2）registerForms()
             用于表单的注册新建。
             一般地，表单统一是在本函数当中新建，可以新建多个表单，当然在其他地方也是可以的。

	public function registerForms()
	{
		$this->form('form_edit')
		     ->init(
		     	 //表单的action
		         ocUrl('/admin/item/update'), 
		         //表单的其他属性
			     array(                       
			     	'enctype' => "multipart/form-data",
					'onsubmit' => 'return validateForm(this)'
			     )
		     );
	}

   
     （3）display()
              用于在表单提交之前的模板输出。没有表单提交时，都会路由到该方法执行。
              因此，如果要输出html，就在这里写输出前的逻辑处理代码，包括表单的输出。当表单提交后，一般是在submit()方法中处理。
              如下实例：

	public function display()
	{
		//获取用户信息
		$userId = $this->request->getGet('user_id');
		$user = new UserService();
		$data = $user->getUserInfo($userId);

	    $this->assign('data', $data);
	}


     （4）submit()
             用于表单提交后的处理。
             默认是通过判断是否POST提交数据。否则，请使用isSubmit()函数去判断是否表单提交。参见表单生成第三第2点。

	public function submit()
	{
		//更新用户信息
		$posts = $this->request->getPost();
		$user = new UserService();
		$user->editUser($posts);
	}


     （5）isSubmit()
             用于手动判断表单是否提交。
             若Action类中没有该方法函数，默认是使用框架的Request::isPost()来检测的；如果不合适，可以新建该函数去检测，然后return true/false即可。
             如下实例：

	public function isSubmit()
	{
	    //重写表单是否提交的判断
		return $this->request->isPost() && array_key_exists('submit_button_name', $_POST);
	}


     （6）ajax()
             用于ajax提交的后台代码处理。框架中使用Request::isAjax()判断是否是ajax提交的。
             如下实例：

public function ajax()
{
	//更新用户信息
	$posts = $this->request->getPost();
	$user = new UserService();
	$user->editUser($posts);

	return true;
}


     （7）after()（后置事件）
             用于动作执行完成以后的处理。display()、submit()和ajax()执行完以后会自动执行。

	public function after()
	{
		//这里写Action执行完以后的代码
	}


             使用上述7个函数，您的代码将变得更加有条理。其中after()、isSubmit()一般要手动添加。

      3、结构控制函数的执行顺序
             上面几个结构型函数按优先级从高到低的执行顺序：
             __action() -> registerForms()-> isSubmit() -> display()或submit()或ajax() -> after()
